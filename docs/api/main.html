149    def process_image(self):
150        """
151        Виконує інференс на завантаженому статичному зображенні.
152
153        Метод перевіряє наявність завантаженої моделі та зображення. Потім передає
154        зображення до моделі YOLO, отримує результати (координати рамок, класи об'єктів),
155        малює їх на зображенні та відображає у правому вікні інтерфейсу.
156        Також викликає метод логування для виведення текстової статистики.
157        """
158        # Запускаємо нейромережу для аналізу завантаженого фото
159        if not self.model:
160            messagebox.showerror("Помилка", "Спочатку завантажте модель (Load Model)!")
161            return
162        if not self.image_path:
163            messagebox.showinfo("Інфо", "Спочатку завантажте фото!")
164            return
165
166        results = self.model.predict(self.image_path)
167        res_plot = results[0].plot()
168
169        # Зберігаємо результат в пам'яті на випадок натискання кнопки "Save"

171
172        # Перетворюємо колір з OpenCV (BGR) у формат для інтерфейсу (RGB)
173        res_rgb = cv2.cvtColor(res_plot, cv2.COLOR_BGR2RGB)
174        res_pil = Image.fromarray(res_rgb)
175        res_pil.thumbnail((400, 400))
176
177        self.photo_after = ImageTk.PhotoImage(res_pil)
178        self.canvas_after.create_image(200, 200, image=self.photo_after)

180        self.log_results(results[0])

226    def video_loop(self):
227        """
228        Основний цикл обробки відеопотоку в реальному часі.
229
230        Метод рекурсивно викликається через метод after() інтерфейсу Tkinter.
231        Він зчитує поточний кадр з об'єкта cv2.VideoCapture, передає його в модель YOLO
232        для детекції (зі зменшеним розміром зображення imgsz=320 для підвищення FPS).
233        Оригінальний та оброблений кадри конвертуються у формат, сумісний з Tkinter,
234        і відображаються на екрані. Якщо встановлено прапорець `is_paused`,
235        обробка кадру тимчасово пропускається.
236        """
237        # Основний цикл програми: читаємо кадр, обробляємо його YOLO та оновлюємо екран
238        if not self.is_running or not self.cap or not self.cap.isOpened():
239            return
240
241        if self.is_paused:
242            self.root.after(100, self.video_loop)
243            return
244
245        ret, frame = self.cap.read()
246        if ret:
247            # Обробка кадру моделлю (використовуємо менший розмір для швидкості)
248            results = self.model(frame, imgsz=320, conf=0.25, verbose=False)  # type: ignore
249            annotated_frame = results[0].plot()
250            self.last_frame_to_save = annotated_frame
251
252            # Оновлюємо ліве (оригінал) та праве (результат) вікна
253            frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
254            self.photo_in_ref = ImageTk.PhotoImage(self.resize_for_canvas(frame_rgb))
255            self.canvas_before.create_image(200, 200, image=self.photo_in_ref)
256
257            annotated_rgb = cv2.cvtColor(annotated_frame, cv2.COLOR_BGR2RGB)
258            self.photo_out_ref = ImageTk.PhotoImage(self.resize_for_canvas(annotated_rgb))
259            self.canvas_after.create_image(200, 200, image=self.photo_out_ref)
260
261            # Рекурсивно викликаємо функцію для наступного кадру
262            self.root.after(1, self.video_loop)
263        else:
264            self.stop_video()

266    def save_results(self):
267        """
268        Зберігає поточний кадр із розпізнаними об'єктами та текстовий звіт на диск.
269
270        Створює унікальні імена файлів на основі поточного часу (формат YYYY-MM-DD_HH-MM-SS).
271        Зберігає зображення (останній оброблений кадр) у форматі .jpg та текстовий
272        файл .txt зі статистикою розпізнавання (час інференсу, класи, впевненість).
273        Усі файли автоматично зберігаються в директорію `saved_results/`.
274        """
275        # Зберігаємо скріншот з рамками та текстовий звіт у папку результатів
276        if self.last_frame_to_save is None:
277            messagebox.showwarning("Warning", "Немає результату для збереження!")
278            return
279
280        now = datetime.datetime.now()
281        timestamp = now.strftime("%Y-%m-%d_%H-%M-%S")
282        base_filename = f"saved_results/result_{timestamp}"
283
284        # Записуємо зображення
285        img_filename = base_filename + ".jpg"
286        cv2.imwrite(img_filename, self.last_frame_to_save)
287
288        # Записуємо текстовий лог
289        txt_filename = base_filename + ".txt"
290        log_content = self.text_results.get(1.0, tk.END).strip()
291
292        with open(txt_filename, "w", encoding="utf-8") as f:
293            f.write(f"Date/Time: {now.strftime('%Y-%m-%d %H:%M:%S')}\n")
294            f.write(f"Image File: {img_filename}\n")
295            f.write("-" * 20 + "\n")
296            f.write("DETECTED OBJECTS:\n")
297            f.write(log_content + "\n")
298
299        messagebox.showinfo("Saved", f"Файли збережено:\n1. {img_filename}\n2. {txt_filename}")

330    def log_results(self, result):
331        """Формує та виводить текстову статистику (швидкість, класи, впевненість) у вікно логів."""
332        # Отримуємо назви та впевненість для кожного знайденого об'єкта
333        self.text_results.delete(1.0, tk.END)
334        if len(result.boxes) == 0:
335            self.text_results.insert(tk.END, "No objects detected.")
336            return
337
338        self.text_results.insert(tk.END, f"Speed: {result.speed['inference']:.1f}ms\n")
339        self.text_results.insert(tk.END, "-" * 20 + "\n")
340
341        for box in result.boxes:
342            cls = int(box.cls[0])
343            conf = float(box.conf[0])
344            name = result.names[cls]
345            self.text_results.insert(tk.END, f"{name}: {conf * 100:.1f}%\n")

355def format_confidence(conf_value: float) -> str:
356    """
357    Форматує значення впевненості нейромережі у відсотки.
358    Це приклад Test-Driven Documentation (живої документації).
359
360    Приклади використання:
361    >>> format_confidence(0.8532)
362    '85.3%'
363    >>> format_confidence(0.9999)
364    '100.0%'
365    >>> format_confidence(0.12)
366    '12.0%'
367    """
368    return f"{conf_value * 100:.1f}%"

227    def video_loop(self):
228        """
229        Основний цикл обробки відеопотоку в реальному часі.
230
231        Метод рекурсивно викликається через метод after() інтерфейсу Tkinter.
232        Він зчитує поточний кадр з об'єкта cv2.VideoCapture, передає його в модель YOLO
233        для детекції (зі зменшеним розміром зображення imgsz=320 для підвищення FPS).
234        Оригінальний та оброблений кадри конвертуються у формат, сумісний з Tkinter,
235        і відображаються на екрані. Якщо встановлено прапорець `is_paused`,
236        обробка кадру тимчасово пропускається.
237        """
238        # Основний цикл програми: читаємо кадр, обробляємо його YOLO та оновлюємо екран
239        if not self.is_running or not self.cap or not self.cap.isOpened():
240            return
241
242        if self.is_paused:
243            self.root.after(100, self.video_loop)
244            return
245
246        ret, frame = self.cap.read()
247        if ret:
248            # Обробка кадру моделлю (використовуємо менший розмір для швидкості)
249            results = self.model(frame, imgsz=320, conf=0.25, verbose=False)  # type: ignore
250            annotated_frame = results[0].plot()
251            self.last_frame_to_save = annotated_frame
252
253            # Оновлюємо ліве (оригінал) та праве (результат) вікна
254            frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
255            self.photo_in_ref = ImageTk.PhotoImage(self.resize_for_canvas(frame_rgb))
256            self.canvas_before.create_image(200, 200, image=self.photo_in_ref)
257
258            annotated_rgb = cv2.cvtColor(annotated_frame, cv2.COLOR_BGR2RGB)
259            self.photo_out_ref = ImageTk.PhotoImage(self.resize_for_canvas(annotated_rgb))
260            self.canvas_after.create_image(200, 200, image=self.photo_out_ref)
261
262            # Рекурсивно викликаємо функцію для наступного кадру
263            self.root.after(1, self.video_loop)
264        else:
265            self.stop_video()

267    def save_results(self):
268        """
269        Зберігає поточний кадр із розпізнаними об'єктами та текстовий звіт на диск.
270
271        Створює унікальні імена файлів на основі поточного часу (формат YYYY-MM-DD_HH-MM-SS).
272        Зберігає зображення (останній оброблений кадр) у форматі .jpg та текстовий
273        файл .txt зі статистикою розпізнавання (час інференсу, класи, впевненість).
274        Усі файли автоматично зберігаються в директорію `saved_results/`.
275        """
276        # Зберігаємо скріншот з рамками та текстовий звіт у папку результатів
277        if self.last_frame_to_save is None:
278            messagebox.showwarning("Warning", "Немає результату для збереження!")
279            return
280
281        now = datetime.datetime.now()
282        timestamp = now.strftime("%Y-%m-%d_%H-%M-%S")
283        base_filename = f"saved_results/result_{timestamp}"
284
285        # Записуємо зображення
286        img_filename = base_filename + ".jpg"
287        cv2.imwrite(img_filename, self.last_frame_to_save)
288
289        # Записуємо текстовий лог
290        txt_filename = base_filename + ".txt"
291        log_content = self.text_results.get(1.0, tk.END).strip()
292
293        with open(txt_filename, "w", encoding="utf-8") as f:
294            f.write(f"Date/Time: {now.strftime('%Y-%m-%d %H:%M:%S')}\n")
295            f.write(f"Image File: {img_filename}\n")
296            f.write("-" * 20 + "\n")
297            f.write("DETECTED OBJECTS:\n")
298            f.write(log_content + "\n")
299
300        messagebox.showinfo("Saved", f"Файли збережено:\n1. {img_filename}\n2. {txt_filename}")
